def wgs84_to_web_mercator(lon, lat):
    """Converts decimal longitude/latitude to Web Mercator format"""
    k = 6378137
    x_center = lon * (k * np.pi/180.0)
    y_center = np.log(np.tan((90 + lat) * np.pi/360.0)) * k
    return x_center, y_center
